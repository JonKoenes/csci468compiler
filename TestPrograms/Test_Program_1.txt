	function factorial( b: integer):integer;
	var x: integer;
	begin
		if b = 0 then
			factorial := 1
		else
			factorial := b * factorial(b-1);
	end;
	
	procedure listNum( start: integer);
	var finish: integer;
	begin
		finish := -50;
		while start > finish do
		begin
			write(start, ' ');
			start := start - 1;
		end;

	end;
